  ```typescript
  import type { Task } from "@/services/task-service"

  const PRIORITY_RANK: Record<string, number> = { critical: 4, high: 3, medium: 2, low: 1 }

  /** Smart default sort: priority desc → due asc (nulls last) → title asc */
  export function smartSort(a: Task, b: Task): number {
    const pa = PRIORITY_RANK[a.priority] ?? 0
    const pb = PRIORITY_RANK[b.priority] ?? 0
    if (pa !== pb) return pb - pa
    const da = a.due ? new Date(a.due).getTime() : Number.POSITIVE_INFINITY
    const db = b.due ? new Date(b.due).getTime() : Number.POSITIVE_INFINITY
    if (da !== db) return da - db
    return a.title.localeCompare(b.title)
  }
  ```

  **Step 3 — Create `Frontend2/lib/my-tasks/due-bucket.ts`:**
  ```typescript
  export type DueBucket = "overdue" | "today" | "this_week" | "later" | "none"

  export function dueBucket(iso: string | null, nowRef?: Date): DueBucket {
    if (!iso) return "none"
    const now = nowRef ?? new Date()
    const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate()).getTime()
    const due = new Date(iso).getTime()
    if (due < todayStart) return "overdue"
    const oneDay = 86400000
    if (due < todayStart + oneDay) return "today"
    if (due < todayStart + 7 * oneDay) return "this_week"
    return "later"
  }
  ```

  **Step 5 — Create `Frontend2/hooks/use-my-tasks-store.ts`:**
  ```typescript
  "use client"
  import * as React from "react"
  import type { Task } from "@/services/task-service"

  const STORE_KEY = "spms.myTasksStore"

  export interface MyTasksStore {
    overrides: Record<string, Partial<Task>>
    extras: Task[]
    starred: number[]
    completedAt: Record<number, string>
  }

  const DEFAULT: MyTasksStore = { overrides: {}, extras: [], starred: [], completedAt: {} }

  function load(): MyTasksStore {
    if (typeof window === "undefined") return DEFAULT
    try {
      const v = window.localStorage.getItem(STORE_KEY)
      if (!v) return DEFAULT
      const parsed = JSON.parse(v)
      return { ...DEFAULT, ...parsed }
    } catch { return DEFAULT }
  }

  function save(s: MyTasksStore): void {
    if (typeof window === "undefined") return
    try { window.localStorage.setItem(STORE_KEY, JSON.stringify(s)) } catch { /* ignore */ }
  }

  export function useMyTasksStore(): [MyTasksStore, (u: (prev: MyTasksStore) => MyTasksStore) => void] {
    const [store, setStore] = React.useState<MyTasksStore>(DEFAULT)
    const hydrated = React.useRef(false)

    React.useEffect(() => {
      setStore(load())
      hydrated.current = true
    }, [])

    const update = React.useCallback((u: (prev: MyTasksStore) => MyTasksStore) => {
      setStore(prev => {
        const next = u(prev)
        if (hydrated.current) save(next)
        return next
      })
    }, [])

    return [store, update]
  }
  ```

  ```tsx
  "use client"
  import { Card } from "@/components/primitives"
  import type { Task } from "@/services/task-service"
  import { TaskRow } from "./task-row"
  import { useApp } from "@/context/app-context"
  import { dueBucket } from "@/lib/my-tasks/due-bucket"
  import type { GroupBy } from "./task-filter-bar"

  interface Props {
    tasks: Task[]
    groupBy: GroupBy
    starred: number[]
    onToggleStar: (id: number) => void
    compact?: boolean
    projectsByKey?: Map<number, string>
  }

  export function TaskGroupList({ tasks, groupBy, starred, onToggleStar, compact, projectsByKey }: Props) {
    const { language: lang } = useApp()
    if (tasks.length === 0) {
      return (
        <div style={{ padding: 24, textAlign: "center", color: "var(--fg-subtle)", fontSize: 12.5 }}>
          {lang === "tr" ? "Görev bulunamadı" : "No tasks found"}
        </div>
      )
    }

    if (groupBy === "none") {
      return (
        <Card padding={0}>
          {tasks.map(t => (
            <TaskRow key={t.id} task={t} compact={compact} starred={starred.includes(t.id)} onToggleStar={() => onToggleStar(t.id)}
              showProject={!!projectsByKey} projectKey={projectsByKey?.get(t.projectId)} />
          ))}
        </Card>
      )
    }

    // Group
    const groups = new Map<string, Task[]>()
    tasks.forEach(t => {
      let k: string
      if (groupBy === "project") k = projectsByKey?.get(t.projectId) ?? `#${t.projectId}`
      else if (groupBy === "status") k = t.status
      else if (groupBy === "priority") k = t.priority
      else k = dueBucket(t.due)
      if (!groups.has(k)) groups.set(k, [])
      groups.get(k)!.push(t)
    })

    return (
      <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
        {Array.from(groups.entries()).map(([k, list]) => (
          <Card key={k} padding={0}>
            <div style={{ padding: "8px 12px", borderBottom: "1px solid var(--border)", fontSize: 11, fontWeight: 600, color: "var(--fg-subtle)", textTransform: "uppercase", letterSpacing: 0.4 }}>
              {k} · {list.length}
            </div>
            {list.map(t => (
              <TaskRow key={t.id} task={t} compact={compact} starred={starred.includes(t.id)} onToggleStar={() => onToggleStar(t.id)}
                showProject={groupBy !== "project" && !!projectsByKey} projectKey={projectsByKey?.get(t.projectId)} />
            ))}
          </Card>
        ))}
      </div>
    )
  }
  ```

  ```tsx
    const { data: tasks = [] } = useMyTasks()
    const { data: projects = [] } = useProjects()
    const [store, setStore] = useMyTasksStore()
    const [view, setView] = React.useState<ViewId>(defaultView)
    const [search, setSearch] = React.useState("")
    const [groupBy, setGroupBy] = React.useState<GroupBy>("none")

    const projectsByKey = React.useMemo(() => {
      const m = new Map<number, string>()
      projects.forEach(p => m.set(p.id, p.key))
      return m
    }, [projects])

    const toggleStar = (id: number) => {
      setStore(prev => ({
        ...prev,
        starred: prev.starred.includes(id) ? prev.starred.filter(x => x !== id) : [...prev.starred, id],
      }))
    }

    // Apply view filter
    const viewFiltered = React.useMemo(() => {
      return tasks.filter(t => {
        const bucket = dueBucket(t.due)
        switch (view) {
          case "today": return bucket === "today" && t.status !== "done"
          case "overdue": return bucket === "overdue" && t.status !== "done"
          case "upcoming": return (bucket === "this_week" || bucket === "later") && t.status !== "done"
          case "starred": return store.starred.includes(t.id)
          case "done": return t.status === "done"
          case "all": return t.status !== "done"
        }
        return true
      })
    }, [tasks, view, store.starred])

    const searched = React.useMemo(() => {
      if (!search.trim()) return viewFiltered
      const q = search.toLowerCase()
      return viewFiltered.filter(t => t.title.toLowerCase().includes(q) || t.key.toLowerCase().includes(q))
    }, [viewFiltered, search])

    const sorted = React.useMemo(() => [...searched].sort(smartSort), [searched])

    // View counts
    const counts = React.useMemo<Record<ViewId, number>>(() => {
      const c: Record<ViewId, number> = { today: 0, overdue: 0, upcoming: 0, starred: 0, all: 0, done: 0 }
      tasks.forEach(t => {
        const bucket = dueBucket(t.due)
        if (t.status === "done") c.done++
        else {
          c.all++
          if (bucket === "today") c.today++
          if (bucket === "overdue") c.overdue++
          if (bucket === "this_week" || bucket === "later") c.upcoming++
        }
        if (store.starred.includes(t.id)) c.starred++
      })
      return c
    }, [tasks, store.starred])

    return (
      <div style={{ display: "flex", flexDirection: "column", gap: compact ? 8 : 16 }}>
        {!hideHeader && !compact && (
          <div>
            <h1 style={{ fontSize: 24, fontWeight: 600, letterSpacing: -0.6, margin: 0 }}>
              {lang === "tr" ? "Görevlerim" : "My Tasks"}
            </h1>
            <div style={{ fontSize: 13, color: "var(--fg-muted)", marginTop: 4 }}>
              {tasks.length} {lang === "tr" ? "görev" : "tasks"}
            </div>
          </div>
        )}

        {!compact && <SavedViewsTabs value={view} onChange={setView} counts={counts} />}

        <TaskFilterBar
          search={search} onSearch={setSearch}
          groupBy={groupBy} onGroupByChange={setGroupBy}
        />

        <TaskGroupList
          tasks={sorted}
          groupBy={groupBy}
          starred={store.starred}
          onToggleStar={toggleStar}
          compact={compact}
          projectsByKey={projectsByKey}
        />
      </div>
    )
  }
  ```
