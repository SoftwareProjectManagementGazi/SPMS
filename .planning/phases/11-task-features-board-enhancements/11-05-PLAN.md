  ```tsx
  export function BoardCard({ task, columnId, projectId, densityMode, phaseNodes, enablePhaseBadge }: BoardCardProps) {

      data: { columnId, task },

    const phaseName = enablePhaseBadge && task.phaseId

          {task.type === "bug" && <Bug size={12} color="var(--priority-critical)" />}
  ```

  **Step 1 — Create `Frontend2/lib/dnd/board-dnd.ts`:**
  ```typescript
  import type { Task } from "@/services/task-service"

  export interface BoardColumnInfo {
    id: string
    wipLimit: number
    taskCount: number // current count BEFORE drop
  }

  export interface DragResult {
    moved: boolean
    wipExceeded: boolean
  }

  /** Pure handler used by BoardTab onDragEnd — separated for unit testing. */
  export function handleBoardDragEnd(params: {
    taskId: number
    sourceColumnId: string
    targetColumnId: string
    targetColumn: BoardColumnInfo | null
  }): DragResult {
    const { sourceColumnId, targetColumnId, targetColumn } = params
    if (!targetColumnId || sourceColumnId === targetColumnId) {
      return { moved: false, wipExceeded: false }
    }
    const projectedCount = (targetColumn?.taskCount ?? 0) + 1
    const wipExceeded = !!(targetColumn?.wipLimit && targetColumn.wipLimit > 0 && projectedCount > targetColumn.wipLimit)
    // D-20 authoritative: Warn + Allow (soft). Move proceeds regardless of wipExceeded.
    return { moved: true, wipExceeded }
  }
  ```

  ```tsx
  "use client"
  import * as React from "react"
  import { useQuery, useQueryClient } from "@tanstack/react-query"
  import { apiClient } from "@/lib/api-client"
  import { ProjectDnDProvider } from "@/lib/dnd/dnd-provider"
  import { useTasks, useMoveTask } from "@/hooks/use-tasks"
  import { showToast } from "@/components/toast"
  import { useApp } from "@/context/app-context"
  import { BoardColumn } from "./board-column"
  import { BoardToolbar } from "./board-toolbar"
  import { BoardCardGhost } from "./board-card"
  import { useProjectDetail } from "./project-detail-context"
  import { handleBoardDragEnd } from "@/lib/dnd/board-dnd"
  import type { Project } from "@/services/project-service"
  import type { Task } from "@/services/task-service"

  interface ColumnMeta { id: number; name: string; order_index: number; wip_limit: number; task_count: number }

  function useColumns(projectId: number) {
    return useQuery({
      queryKey: ["columns", projectId],
      queryFn: async () => (await apiClient.get<ColumnMeta[]>(`/projects/${projectId}/columns`)).data,
    })
  }

  export function BoardTab({ project }: { project: Project }) {
    const { language: lang } = useApp()
    const pd = useProjectDetail()
    const { data: tasks = [] } = useTasks(project.id)
    const { data: columns = [] } = useColumns(project.id)
    const moveTask = useMoveTask(project.id)

    const phaseEnabled = !!((project.processConfig as { enable_phase_assignment?: boolean } | null)?.enable_phase_assignment)
    const phaseNodes = ((project.processConfig as { workflow?: { nodes?: Array<{ id: string; name: string }> } } | null)?.workflow?.nodes) ?? []

    // Apply filters (search, phase)
    const filteredTasks = React.useMemo(() => {
      return tasks.filter(t => {
        if (pd.searchQuery.trim()) {
          const q = pd.searchQuery.toLowerCase()
          if (!t.title.toLowerCase().includes(q) && !t.key.toLowerCase().includes(q)) return false
        }
        if (pd.phaseFilter && t.phaseId !== pd.phaseFilter) return false
        return true
      })
    }, [tasks, pd.searchQuery, pd.phaseFilter])

    // Group by column (status)
    const columnNames = columns.length > 0 ? columns.map(c => c.name) : project.columns
    const wipLimits = new Map<string, number>(columns.map(c => [c.name.toLowerCase(), c.wip_limit]))
    const grouped: Record<string, Task[]> = {}
    columnNames.forEach(cn => { grouped[cn] = [] })
    filteredTasks.forEach(t => {
      const key = columnNames.find(cn => cn.toLowerCase() === t.status.toLowerCase())
        ?? columnNames[0] ?? t.status
      if (!grouped[key]) grouped[key] = []
      grouped[key].push(t)
    })

    const handleDropped = React.useCallback((taskId: number, sourceColumnId: string, targetColumnId: string) => {
      const targetColumnName = targetColumnId
      const targetInfo = {
        id: targetColumnName,
        wipLimit: wipLimits.get(targetColumnName.toLowerCase()) ?? 0,
        taskCount: (grouped[targetColumnName] ?? []).length,
      }
      const result = handleBoardDragEnd({ taskId, sourceColumnId, targetColumnId, targetColumn: targetInfo })
      if (!result.moved) return
      if (result.wipExceeded) {
        showToast({
          variant: "warning",
          message: lang === "tr" ? "WIP limiti aşıldı — kolonda uyarı gösteriliyor" : "WIP limit exceeded — warning shown",
        })
      }
      moveTask.mutate({ id: taskId, status: targetColumnName.toLowerCase() })
    }, [grouped, wipLimits, moveTask, lang])

    const ghostTask = (taskId: number | null) => {
      if (taskId == null) return null
      const t = tasks.find(x => x.id === taskId)
      return t ? <BoardCardGhost task={t} /> : null
    }

    return (
      <div style={{ display: "flex", flexDirection: "column", gap: 12, height: "100%" }}>
        <BoardToolbar project={project} />
        <ProjectDnDProvider projectId={project.id} onTaskDropped={handleDropped} renderGhost={ghostTask}>
          <div style={{
            display: "grid",
            gridTemplateColumns: `repeat(${columnNames.length}, minmax(260px, 1fr))`,
            gap: 12,
            flex: 1,
            minHeight: 0,
          }}>
            {columnNames.map(cn => (
              <BoardColumn
                key={cn}
                columnId={cn}
                columnName={cn}
                wipLimit={wipLimits.get(cn.toLowerCase()) ?? 0}
                tasks={grouped[cn] ?? []}
                projectId={project.id}
                densityMode={pd.densityMode}
                phaseNodes={phaseNodes}
                enablePhaseBadge={phaseEnabled}
              />
            ))}
          </div>
        </ProjectDnDProvider>
      </div>
    )
  }
  ```
