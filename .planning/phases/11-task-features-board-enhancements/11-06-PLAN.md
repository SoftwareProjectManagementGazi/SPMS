  ```tsx
  "use client"
  import * as React from "react"
  import { useDraggable } from "@dnd-kit/core"
  import { CSS } from "@dnd-kit/utilities"
  import { useRouter } from "next/navigation"
  import { Avatar, PriorityChip } from "@/components/primitives"
  import type { Task } from "@/services/task-service"

  const BACKLOG_COLUMN_ID = "__backlog__"

  interface BacklogTaskRowProps {
    task: Task
    projectId: number
  }

  export function BacklogTaskRow({ task, projectId }: BacklogTaskRowProps) {
    const router = useRouter()
    const { attributes, listeners, setNodeRef, transform, isDragging } = useDraggable({
      id: task.id,
      data: { columnId: BACKLOG_COLUMN_ID, task },
    })

    return (
      <div
        ref={setNodeRef}
        {...attributes}
        {...listeners}
        style={{
          display: "flex", alignItems: "center", gap: 8,
          padding: "8px 12px",
          borderBottom: "1px solid var(--border)",
          borderLeft: `3px solid var(--priority-${task.priority})`,
          background: isDragging ? "var(--surface-2)" : "transparent",
          cursor: "grab",
          transform: CSS.Transform.toString(transform),
          opacity: isDragging ? 0.5 : 1,
        }}
        onClick={(e) => {
          if (isDragging) return
          // click navigates to detail (same semantics as board card per D-23)
          router.push(`/projects/${projectId}/tasks/${task.id}`)
          e.stopPropagation()
        }}
      >
        <span style={{ fontFamily: "var(--font-mono)", fontSize: 10.5, color: "var(--fg-subtle)", letterSpacing: 0.3, minWidth: 60 }}>
          {task.key}
        </span>
        <span style={{ fontSize: 12.5, fontWeight: 500, color: "var(--fg)", flex: 1, overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap" }}>
          {task.title}
        </span>
        <PriorityChip level={task.priority} />
        {task.assigneeId != null && (
          <Avatar user={{ id: task.assigneeId, name: `#${task.assigneeId}`, avColor: (task.assigneeId % 8) + 1 }} size={20} />
        )}
      </div>
    )
  }

  export { BACKLOG_COLUMN_ID }
  ```

  ```tsx
  "use client"
  import * as React from "react"
  import { Search, X } from "lucide-react"
  import { Input, Badge, AlertBanner, Avatar } from "@/components/primitives"
  import { useApp } from "@/context/app-context"
  import { useBacklog } from "@/hooks/use-backlog"
  import type { Project } from "@/services/project-service"
  import type { Task } from "@/services/task-service"
  import { BacklogTaskRow } from "./backlog-task-row"

  type Priority = "low" | "medium" | "high" | "critical"
  const PRIORITIES: Priority[] = ["critical", "high", "medium", "low"]
  const PRIO_LABEL: Record<Priority, { tr: string; en: string }> = {
    critical: { tr: "Kritik", en: "Critical" },
    high: { tr: "Yüksek", en: "High" },
    medium: { tr: "Orta", en: "Medium" },
    low: { tr: "Düşük", en: "Low" },
  }

  const OPEN_KEY = (projectId: number) => `spms.backlog.open.${projectId}`

  export function useBacklogOpenState(projectId: number) {
    // Default closed on first visit (D-14)
    const [open, setOpen] = React.useState<boolean>(false)
    const [narrow, setNarrow] = React.useState<boolean>(false)

    React.useEffect(() => {
      // Hydrate from localStorage after mount (SSR-safe)
      try {
        const stored = localStorage.getItem(OPEN_KEY(projectId))
        if (stored !== null) setOpen(stored === "true")
      } catch { /* ignore */ }
    }, [projectId])

    React.useEffect(() => {
      try { localStorage.setItem(OPEN_KEY(projectId), String(open)) } catch { /* ignore */ }
    }, [projectId, open])

    React.useEffect(() => {
      const onResize = () => setNarrow(window.innerWidth < 1280)
      onResize()
      window.addEventListener("resize", onResize)
      return () => window.removeEventListener("resize", onResize)
    }, [])

    // Per D-54: <1280px auto-closes; toggle still works to force-open
    const effectiveOpen = narrow ? false : open
    return { open, effectiveOpen, narrow, setOpen }
  }

  interface BacklogPanelProps {
    project: Project
    open: boolean
    onClose: () => void
    narrow: boolean
  }

  export function BacklogPanel({ project, open, onClose, narrow }: BacklogPanelProps) {
    const { language: lang } = useApp()
    const { data: tasks = [], isLoading } = useBacklog(project)
    const [search, setSearch] = React.useState("")
    const [activePriorities, setActivePriorities] = React.useState<Priority[]>([])
    const [activeAssignees, setActiveAssignees] = React.useState<number[]>([])

    const togglePriority = (p: Priority) => {
      setActivePriorities(prev => prev.includes(p) ? prev.filter(x => x !== p) : [...prev, p])
    }
    const toggleAssignee = (id: number) => {
      setActiveAssignees(prev => prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id])
    }

    const filtered = React.useMemo<Task[]>(() => {
      return tasks.filter(t => {
        if (search.trim()) {
          const q = search.toLowerCase()
          if (!t.title.toLowerCase().includes(q) && !t.key.toLowerCase().includes(q)) return false
        }
        if (activePriorities.length > 0 && !activePriorities.includes(t.priority)) return false
        if (activeAssignees.length > 0 && (t.assigneeId == null || !activeAssignees.includes(t.assigneeId))) return false
        return true
      })
    }, [tasks, search, activePriorities, activeAssignees])

    // Unique assignees on backlog tasks for the avatar filter
    const uniqueAssignees = React.useMemo(() => {
      const ids = new Set<number>()
      tasks.forEach(t => { if (t.assigneeId != null) ids.add(t.assigneeId) })
      return Array.from(ids)
    }, [tasks])

    // Resolve backlog definition label for the header pill
    const cfg = (project.processConfig ?? {}) as { backlog_definition?: string }
    const defLabel: Record<string, { tr: string; en: string }> = {
      cycle_null: { tr: "Döngü'e atanmamış", en: "Not in cycle" },
      leftmost_column: { tr: "İlk kolon", en: "Leftmost column" },
      phase_null_or_first: { tr: "Faza atanmamış", en: "Not in phase" },
      custom: { tr: "Özel", en: "Custom" },
    }
    const label = defLabel[cfg.backlog_definition ?? "cycle_null"] ?? { tr: "Backlog", en: "Backlog" }

    if (!open) return null

    return (
      <aside
        aria-label="Backlog panel"
        style={{
          width: 300, flexShrink: 0,
          background: "var(--surface)",
          borderRight: "1px solid var(--border)",
          display: "flex", flexDirection: "column",
          height: "100%",
          transition: "width 180ms ease",
        }}
      >
        {/* Header */}
        <div style={{ padding: "12px 16px", borderBottom: "1px solid var(--border)", display: "flex", alignItems: "center", gap: 6 }}>
          <div style={{ fontSize: 13, fontWeight: 600, flex: 1 }}>Backlog</div>
          <Badge size="xs" tone="neutral">{lang === "tr" ? label.tr : label.en}</Badge>
          <button onClick={onClose} aria-label="Close" style={{ background: "transparent", border: "none", cursor: "pointer", color: "var(--fg-muted)", padding: 4 }}>
            <X size={14} />
          </button>
        </div>

        {narrow && (
          <div style={{ padding: "8px 12px", background: "var(--surface-2)", fontSize: 11.5, color: "var(--fg-subtle)" }}>
            {lang === "tr" ? "Dar ekranda kısıtlı görünüm" : "Narrow screen — limited view"}
          </div>
        )}

        {/* Toolbar: search + filters */}
        <div style={{ padding: 8, display: "flex", flexDirection: "column", gap: 6, borderBottom: "1px solid var(--border)" }}>
          <Input
            icon={<Search size={12} />}
            placeholder={lang === "tr" ? "Görevlerde ara…" : "Search tasks…"}
            size="sm"
            value={search}
            onChange={e => setSearch(e.target.value)}
          />
          <div style={{ display: "flex", flexWrap: "wrap", gap: 4 }}>
            {PRIORITIES.map(p => {
              const active = activePriorities.includes(p)
              return (
                <button key={p} onClick={() => togglePriority(p)}
                  style={{
                    fontSize: 11, padding: "3px 8px", borderRadius: 999,
                    border: "none", cursor: "pointer",
                    background: active ? `color-mix(in oklch, var(--priority-${p}) 25%, transparent)` : "var(--surface-2)",
                    color: active ? "var(--fg)" : "var(--fg-muted)",
                    boxShadow: active ? `inset 0 0 0 1px var(--priority-${p})` : "inset 0 0 0 1px var(--border)",
                    fontWeight: 500,
                  }}>
                  {PRIO_LABEL[p][lang]}
                </button>
              )
            })}
          </div>
          {uniqueAssignees.length > 0 && (
            <div style={{ display: "flex", flexWrap: "wrap", gap: 4 }}>
              {uniqueAssignees.map(aid => {
                const active = activeAssignees.includes(aid)
                return (
                  <button key={aid} onClick={() => toggleAssignee(aid)}
                    style={{
                      padding: 2, borderRadius: 999,
                      border: "none", cursor: "pointer",
                      background: "transparent",
                      boxShadow: active ? "0 0 0 2px var(--primary)" : "none",
                    }}>
                    <Avatar user={{ id: aid, name: `#${aid}`, avColor: (aid % 8) + 1 }} size={22} />
                  </button>
                )
              })}
            </div>
          )}
        </div>

        {/* Task list */}
        <div style={{ flex: 1, overflowY: "auto" }}>
          {isLoading ? (
            <div style={{ padding: 20, color: "var(--fg-muted)", fontSize: 12, textAlign: "center" }}>
              {lang === "tr" ? "Yükleniyor…" : "Loading…"}
            </div>
          ) : filtered.length === 0 ? (
            <div style={{ padding: 32, textAlign: "center", color: "var(--fg-subtle)" }}>
              <div style={{ fontSize: 13, fontWeight: 600, marginBottom: 6 }}>
                {lang === "tr" ? "Backlog boş" : "Backlog is empty"}
              </div>
              <div style={{ fontSize: 12, color: "var(--fg-muted)" }}>
                {lang === "tr" ? "Filtreleri temizleyin veya görev ekleyin" : "Clear filters or add tasks"}
              </div>
            </div>
          ) : (
            filtered.map(t => <BacklogTaskRow key={t.id} task={t} projectId={project.id} />)
          )}
        </div>

        {/* Footer: count */}
        <div style={{ padding: "8px 12px", borderTop: "1px solid var(--border)", fontFamily: "var(--font-mono)", fontSize: 11, color: "var(--fg-muted)" }}>
          {filtered.length} / {tasks.length} {lang === "tr" ? "görev" : "tasks"}
        </div>
      </aside>
    )
  }
  ```

  ```tsx
      const currentCount = (currentTasks as Array<{ status: string }>).filter(t => t.status.toLowerCase() === targetColumnId.toLowerCase()).length
      const targetInfo: BoardColumnInfo = {
        id: targetColumnId,
        wipLimit: targetCol?.wip_limit ?? 0,
        taskCount: currentCount,
      }
      const result = handleBoardDragEnd({ taskId, sourceColumnId, targetColumnId, targetColumn: targetInfo })
      if (!result.moved) return
      if (result.wipExceeded) {
        showToast({
          variant: "warning",
          message: lang === "tr" ? "WIP limiti aşıldı" : "WIP limit exceeded",
        })
      }
      moveTask.mutate({ id: taskId, status: targetColumnId.toLowerCase() })
      // Cross-container: when source is backlog, the move-task mutation changes
      // the status (making it non-backlog in SCRUM by assignment + Kanban by
      // status) — backlog query invalidation is automatic via onSettled in useMoveTask,
      // which invalidates ['tasks', 'project', projectId]; backlog queryKey is
      // ['tasks', 'backlog', projectId, filter] → also invalidate here.
      qc.invalidateQueries({ queryKey: ["tasks", "backlog", project.id] })
    }, [columns, currentTasks, moveTask, project.id, qc, lang])

    const renderGhost = React.useCallback((taskId: number | null) => {
      if (taskId == null) return null
      const t = (currentTasks as Array<{ id: number; key: string; title: string; priority: "low"|"medium"|"high"|"critical" }>).find(x => x.id === taskId)
      return t ? <BoardCardGhost task={t as never} /> : null
    }, [currentTasks])

    return (
      <ProjectDetailProvider projectId={project.id}>
        <ProjectDnDProvider projectId={project.id} onTaskDropped={handleDropped} renderGhost={renderGhost}>
          <div style={{ display: "flex", flexDirection: "row", gap: 0, flex: 1, minHeight: 0, height: "100%" }}>
            {/* Left edge toggle */}
            <div style={{ display: "flex", alignItems: "center", paddingRight: 4 }}>
              <BacklogToggle open={effectiveOpen} onToggle={() => setOpen(!open)} />
            </div>

            {/* Backlog panel (conditionally rendered) */}
            <BacklogPanel
              project={project}
              open={effectiveOpen}
              narrow={narrow}
              onClose={() => setOpen(false)}
            />

            {/* Tab content */}
            <div style={{ flex: 1, minWidth: 0, display: "flex", flexDirection: "column", gap: 16, marginLeft: effectiveOpen ? 12 : 0 }}>
              <Tabs tabs={tabs} active={tab} onChange={(id: string) => setTab(id as TabId)} size="md" />
              <div style={{ flex: 1, minHeight: 0 }}>
                {tab === "board" && <BoardTab project={project} />}
                {tab === "list" && <TabPlaceholder label={lang === "tr" ? "Liste sekmesi — Plan 11-07" : "List — Plan 11-07"} />}
                {tab === "timeline" && <TabPlaceholder label={lang === "tr" ? "Zaman Çizelgesi — Plan 11-07" : "Timeline — Plan 11-07"} />}
                {tab === "calendar" && <TabPlaceholder label={lang === "tr" ? "Takvim — Plan 11-07" : "Calendar — Plan 11-07"} />}
                {tab === "activity" && <ActivityStubTab />}
                {tab === "lifecycle" && <LifecycleStubTab />}
                {tab === "members" && <MembersTab project={project} />}
                {tab === "settings" && <SettingsTabLazy project={project} isArchived={isArchived} />}
              </div>
            </div>
          </div>
        </ProjectDnDProvider>
      </ProjectDetailProvider>
    )
  }

  function TabPlaceholder({ label }: { label: string }) {
    return (
      <div style={{ padding: 40, textAlign: "center", color: "var(--fg-subtle)", fontSize: 12.5, border: "1px dashed var(--border)", borderRadius: "var(--radius-sm)" }}>
        {label}
      </div>
    )
  }

  const SettingsTabLazy = React.lazy(async () => ({
    default: (await import("./settings-tab")).SettingsTab,
  }))
  ```

  ```tsx
  // ... imports, remove: ProjectDnDProvider, handleBoardDragEnd, BoardCardGhost if unused locally
  "use client"
  import * as React from "react"
  import { useQuery } from "@tanstack/react-query"
  import { apiClient } from "@/lib/api-client"
  import { useTasks } from "@/hooks/use-tasks"
  import { useApp } from "@/context/app-context"
  import { BoardColumn } from "./board-column"
  import { BoardToolbar } from "./board-toolbar"
  import { useProjectDetail } from "./project-detail-context"
  import type { Project } from "@/services/project-service"
  import type { Task } from "@/services/task-service"

  interface ColumnMeta { id: number; name: string; order_index: number; wip_limit: number; task_count: number }

  function useColumns(projectId: number) {
    return useQuery({
      queryKey: ["columns", projectId],
      queryFn: async () => (await apiClient.get<ColumnMeta[]>(`/projects/${projectId}/columns`)).data,
    })
  }

  export function BoardTab({ project }: { project: Project }) {
    const pd = useProjectDetail()
    const { data: tasks = [] } = useTasks(project.id)
    const { data: columns = [] } = useColumns(project.id)

    const phaseEnabled = !!((project.processConfig as { enable_phase_assignment?: boolean } | null)?.enable_phase_assignment)
    const phaseNodes = ((project.processConfig as { workflow?: { nodes?: Array<{ id: string; name: string }> } } | null)?.workflow?.nodes) ?? []

    const filteredTasks = React.useMemo(() => {
      return tasks.filter(t => {
        if (pd.searchQuery.trim()) {
          const q = pd.searchQuery.toLowerCase()
          if (!t.title.toLowerCase().includes(q) && !t.key.toLowerCase().includes(q)) return false
        }
        if (pd.phaseFilter && t.phaseId !== pd.phaseFilter) return false
        return true
      })
    }, [tasks, pd.searchQuery, pd.phaseFilter])

    const columnNames = columns.length > 0 ? columns.map(c => c.name) : project.columns
    const wipLimits = new Map<string, number>(columns.map(c => [c.name.toLowerCase(), c.wip_limit]))
    const grouped: Record<string, Task[]> = {}
    columnNames.forEach(cn => { grouped[cn] = [] })
    filteredTasks.forEach(t => {
      const key = columnNames.find(cn => cn.toLowerCase() === t.status.toLowerCase()) ?? columnNames[0] ?? t.status
      if (!grouped[key]) grouped[key] = []
      grouped[key].push(t)
    })

    return (
      <div style={{ display: "flex", flexDirection: "column", gap: 12, height: "100%" }}>
        <BoardToolbar project={project} />
        <div style={{
          display: "grid",
          gridTemplateColumns: `repeat(${columnNames.length}, minmax(260px, 1fr))`,
          gap: 12,
          flex: 1,
          minHeight: 0,
        }}>
          {columnNames.map(cn => (
            <BoardColumn
              key={cn}
              columnId={cn}
              columnName={cn}
              wipLimit={wipLimits.get(cn.toLowerCase()) ?? 0}
              tasks={grouped[cn] ?? []}
              projectId={project.id}
              densityMode={pd.densityMode}
              phaseNodes={phaseNodes}
              enablePhaseBadge={phaseEnabled}
            />
          ))}
        </div>
      </div>
    )
  }
  ```
