**create-task-modal.tsx — Similar task warning:**

1. Add debounced search state at top of modal:
```typescript
const [titleQuery, setTitleQuery] = useState('')
const [similarTasks, setSimilarTasks] = useState<Task[]>([])
const STOP_WORDS = new Set(['the','a','an','is','in','on','at','to','for','of','and','or'])

const debouncedTitle = useRef<ReturnType<typeof setTimeout> | null>(null)

const handleTitleChange = (value: string) => {
  form.setValue('title', value)  // or however the title field is managed
  setTitleQuery(value)
  if (debouncedTitle.current) clearTimeout(debouncedTitle.current)
  debouncedTitle.current = setTimeout(async () => {
    const words = value.toLowerCase().split(/\s+/).filter(w => !STOP_WORDS.has(w) && w.length > 2)
    if (words.length === 0 || !projectId) { setSimilarTasks([]); return }
    try {
      const results = await taskService.searchSimilar(projectId, words.join(' '))
      setSimilarTasks(results)
    } catch { setSimilarTasks([]) }
  }, 600)
}
```
