Soft delete filter in repo (Pattern 5 from RESEARCH.md):
```python
# Add to every get query
def _get_base_query(self):
    return select(TaskModel).where(TaskModel.is_deleted == False).options(...)

# Soft delete instead of SQL DELETE
async def delete(self, task_id: int) -> bool:
    stmt = select(TaskModel).where(TaskModel.id == task_id, TaskModel.is_deleted == False)
    result = await self.session.execute(stmt)
    model = result.scalar_one_or_none()
    if model:
        model.is_deleted = True
        model.deleted_at = datetime.utcnow()  # Set explicitly — NOT via onupdate (Pitfall 2)
        await self.session.commit()
        return True
    return False
```
