return useMutation({
  mutationFn: ({ field, value }) => taskService.patchField(taskId, field, value),
  onMutate: async ({ field, value }) => {
    await qc.cancelQueries({ queryKey: ["tasks", taskId] })
    const prev = qc.getQueryData<Task>(["tasks", taskId])
    if (prev) qc.setQueryData<Task>(["tasks", taskId], { ...prev, [field]: value } as Task)
    return { prev }
  },
  onError: (_err, _vars, ctx) => { if (ctx?.prev) qc.setQueryData(["tasks", taskId], ctx.prev) },
  onSettled: () => { qc.invalidateQueries({ queryKey: ["tasks", taskId] }) },
})
