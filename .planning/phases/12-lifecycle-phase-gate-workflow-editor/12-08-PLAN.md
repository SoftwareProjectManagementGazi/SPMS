    ```tsx
    React.useEffect(() => {
      const handler = (e: KeyboardEvent) => {
        const meta = isMac() ? e.metaKey : e.ctrlKey
        // Cmd+Z undo
        if (meta && e.key === 'z' && !e.shiftKey) { e.preventDefault(); const prev = history.undo(workflow); if (prev) setWorkflow(prev) }
        // Cmd+Shift+Z redo
        if (meta && e.key === 'z' && e.shiftKey) { e.preventDefault(); const next = history.redo(workflow); if (next) setWorkflow(next) }
        // Cmd+S save (Plan 12-09 wires the actual save handler — for now, prevent browser default)
        if (meta && e.key === 's') { e.preventDefault(); /* TODO Plan 12-09 */ }
        // N add node at cursor
        if (e.key === 'n' && !meta) { e.preventDefault(); addNodeAtCursor() }
        // Delete remove selection
        if (e.key === 'Delete' || e.key === 'Backspace') { e.preventDefault(); deleteSelection() }
        // Cmd+A select all
        if (meta && e.key === 'a') { e.preventDefault(); selectAll() }
        // F fit view
        if (e.key === 'f' && !meta) { e.preventDefault(); fitView() }
        // Esc deselect + close panel
        if (e.key === 'Escape') { e.preventDefault(); setSelected(null); setContextMenu(null) }
      }
      window.addEventListener('keydown', handler)
      return () => window.removeEventListener('keydown', handler)
    }, [history, workflow])
    ```
