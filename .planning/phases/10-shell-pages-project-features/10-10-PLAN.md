```python
"""Integration tests for activity endpoints.
Covers:
- GET /api/v1/projects/{id}/activity (project-scoped, from Phase 9)
- GET /api/v1/activity (global feed, Phase 10 D-28 addition)
"""
import pytest
from httpx import AsyncClient


@pytest.mark.asyncio
async def test_get_global_activity_returns_200(authenticated_client: AsyncClient):
    """D-28: global activity feed requires authentication, returns paginated items."""
    response = await authenticated_client.get("/api/v1/activity?limit=20&offset=0")
    assert response.status_code == 200
    data = response.json()
    assert "items" in data, "Response must have 'items' key"
    assert "total" in data, "Response must have 'total' key"
    assert isinstance(data["items"], list)
    assert isinstance(data["total"], int)
    assert data["total"] >= 0


@pytest.mark.asyncio
async def test_get_global_activity_default_params(authenticated_client: AsyncClient):
    """GET /activity with no params uses defaults (limit=20, offset=0)."""
    response = await authenticated_client.get("/api/v1/activity")
    assert response.status_code == 200
    data = response.json()
    assert len(data["items"]) <= 20


@pytest.mark.asyncio
async def test_get_global_activity_requires_auth(client: AsyncClient):
    """Unauthenticated request to GET /activity returns 401."""
    response = await client.get("/api/v1/activity")
    assert response.status_code == 401


@pytest.mark.asyncio
async def test_get_global_activity_pagination(authenticated_client: AsyncClient):
    """Offset parameter skips items in the result set."""
    resp_page1 = await authenticated_client.get("/api/v1/activity?limit=5&offset=0")
    resp_page2 = await authenticated_client.get("/api/v1/activity?limit=5&offset=5")
    assert resp_page1.status_code == 200
    assert resp_page2.status_code == 200
    # Pages should have at most 5 items each
    assert len(resp_page1.json()["items"]) <= 5
    assert len(resp_page2.json()["items"]) <= 5
```
