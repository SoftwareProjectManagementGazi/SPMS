    **`Frontend/components/project/gantt-tab.tsx`** — `"use client"` mandatory:

    ```tsx
    "use client"
    import { useEffect, useRef, useState } from 'react'
    import { ParentTask } from '@/lib/types'

    // Import frappe-gantt — SSR-safe because this file is "use client"
    // Parent must wrap with: dynamic(() => import('./gantt-tab'), { ssr: false })

    interface Props {
      tasks: ParentTask[]
    }

    export function GanttTab({ tasks }: Props) {
      const containerRef = useRef<HTMLDivElement>(null)
      const ganttRef = useRef<any>(null)
      const [viewMode, setViewMode] = useState<'Day' | 'Week'>('Day')

      const undatedCount = tasks.filter(t => !t.dueDate).length

      const ganttTasks = tasks
        .filter(t => t.dueDate != null)
        .map(t => ({
          id: t.id,
          name: t.key + ' ' + t.title,
          start: t.createdAt.split('T')[0],  // fallback: no start_date field on ParentTask
          end: t.dueDate!.split('T')[0],
          progress: 0,
          dependencies: '',  // Phase 4: no dependency arrows — task dependency IDs not in ParentTask type
          custom_class: `priority-${t.priority.toLowerCase()}`,
        }))
    ```

    Notes on dependencies field:
    - The `ParentTask` type does NOT currently include a `dependencies` array field.
    - Do NOT add a new API call just for dependency data. Set `dependencies: ''` for all tasks.
    - The CONTEXT says "finish-to-start dependencies rendered as arrows" — this requires the dependency IDs on each task. Without a data field, arrows cannot be drawn. Add a TODO comment and leave dependencies empty. The Gantt will still render bars; arrows are a Phase 7 enhancement pending the ParentTask type update.
    - If task-service.ts already returns dependency data on the task DTO, use it. Check: `TaskResponseDTO` does not have a `dependencies` field. So leave empty.

    ```tsx
      useEffect(() => {
        if (!containerRef.current || ganttTasks.length === 0) return
        // Cleanup previous instance
        containerRef.current.innerHTML = ''
        ganttRef.current = null

        // Dynamic import to handle SSR (belt-and-suspenders)
        import('frappe-gantt').then(({ default: Gantt }) => {
          if (!containerRef.current) return
          ganttRef.current = new Gantt(containerRef.current, ganttTasks, {
            view_mode: viewMode,
            draggable: false,  // read-only per CONTEXT.md
            date_format: 'YYYY-MM-DD',
            custom_popup_html: (task: any) => `
              <div style="padding:8px;background:white;border-radius:6px;box-shadow:0 2px 8px rgba(0,0,0,0.15);min-width:200px;">
                <strong style="font-size:13px;">${task.name}</strong>
                <div style="font-size:11px;color:#666;margin-top:4px;">${task._start} → ${task._end}</div>
                <a href="/tasks/${task.id}" style="font-size:12px;color:#3b82f6;display:block;margin-top:6px;">View full task →</a>
              </div>
            `,
          })
        })

        return () => {
          if (containerRef.current) containerRef.current.innerHTML = ''
          ganttRef.current = null
        }
      }, [ganttTasks, viewMode])

      return (
        <div className="space-y-4">
          <div className="flex items-center justify-between">
            {undatedCount > 0 && (
              <p className="text-sm text-muted-foreground">
                {undatedCount} task{undatedCount !== 1 ? 's' : ''} without dates not shown.
              </p>
            )}
            <div className="flex gap-2 ml-auto">
              <button
                onClick={() => setViewMode('Day')}
                className={`px-3 py-1 text-sm rounded ${viewMode === 'Day' ? 'bg-primary text-primary-foreground' : 'bg-muted'}`}
              >Day</button>
              <button
                onClick={() => setViewMode('Week')}
                className={`px-3 py-1 text-sm rounded ${viewMode === 'Week' ? 'bg-primary text-primary-foreground' : 'bg-muted'}`}
              >Week</button>
            </div>
          </div>
          {ganttTasks.length === 0 ? (
            <div className="py-12 text-center text-muted-foreground">
              No tasks with dates to display on the timeline.
            </div>
          ) : (
            <div className="overflow-x-auto">
              <div ref={containerRef} />
            </div>
          )}
        </div>
      )
    }
    ```
