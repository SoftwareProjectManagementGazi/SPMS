```tsx
"use client"
import * as React from "react"
import { useRouter } from "next/navigation"
import { Folder, CheckSquare, CircleCheck, Edit } from "lucide-react"
import { Card, Avatar, Badge, Button } from "@/components/primitives"
import { useApp } from "@/context/app-context"
import { useAuth } from "@/context/auth-context"

interface ProfileHeaderProps {
  user: {
    id: number
    full_name: string
    email: string
    role?: string | { name: string } | null
    avatar_url?: string | null
  }
  stats: {
    projectsTotal: number
    assignedTasks: number
    completedTasks: number
  }
}

function tonesFromRole(roleName: string): "danger" | "info" | "neutral" {
  const n = roleName.toLowerCase()
  if (n === "admin") return "danger"
  if (n.includes("project manager") || n.includes("manager")) return "info"
  return "neutral"
}

export function ProfileHeader({ user, stats }: ProfileHeaderProps) {
  const { language } = useApp()
  const { user: currentUser } = useAuth()
  const router = useRouter()
  const T = (tr: string, en: string) => language === "tr" ? tr : en
  const isSelf = currentUser?.id === user.id
  const roleName = (typeof user.role === "string" ? user.role : user.role?.name) || ""
  const initials = (user.full_name || "")
    .split(" ").filter(Boolean).slice(0, 2).map(n => n[0] || "").join("").toUpperCase()
  const avatarUser = { initials, avColor: ((user.id || 0) % 8) + 1 }

  return (
    <Card padding={24} className="profile-header-card">
      <div style={{ display: "flex", alignItems: "center", gap: 20 }}>
        <Avatar user={avatarUser} size={64} ring={isSelf} style={{ fontSize: 24 }} />
        <div style={{ flex: 1, minWidth: 0 }}>
          <div style={{ display: "flex", alignItems: "center", gap: 10, flexWrap: "wrap" }}>
            <h1 style={{ fontSize: 22, fontWeight: 600, letterSpacing: -0.4, margin: 0 }}>
              {user.full_name}
            </h1>
            {roleName && <Badge tone={tonesFromRole(roleName)}>{roleName}</Badge>}
            {isSelf && <Badge tone="primary" size="xs">{T("Sen", "You")}</Badge>}
          </div>
          <div style={{ fontSize: 13, color: "var(--fg-muted)", marginTop: 4 }}>
            {user.email}
          </div>
          <div style={{ display: "flex", alignItems: "center", gap: 16, marginTop: 8, fontSize: 12.5, color: "var(--fg-muted)" }}>
            <span style={{ display: "inline-flex", alignItems: "center", gap: 4 }}>
              <Folder size={12} />
              {stats.projectsTotal} {T("proje", "projects")}
            </span>
            <span style={{ display: "inline-flex", alignItems: "center", gap: 4 }}>
              <CheckSquare size={12} />
              {stats.assignedTasks} {T("görev", "tasks")}
            </span>
            <span style={{ display: "inline-flex", alignItems: "center", gap: 4 }}>
              <CircleCheck size={12} />
              {stats.completedTasks} {T("tamamlanan", "completed")}
            </span>
          </div>
        </div>
        {isSelf && (
          <Button
            variant="secondary"
            size="sm"
            onClick={() => router.push("/settings")}
            className="profile-header-edit-button"
          >
            <Edit size={13} />
            {T("Düzenle", "Edit")}
          </Button>
        )}
      </div>
    </Card>
  )
}
```

```tsx
"use client"
import * as React from "react"
import { useQuery } from "@tanstack/react-query"
import { Card, SegmentedControl, Badge, DataState } from "@/components/primitives"
import { TaskRow } from "@/components/my-tasks/task-row"
import { useApp } from "@/context/app-context"
import { profileService } from "@/services/profile-service"

type TaskFilter = "active" | "completed" | "all"

interface ProfileTasksTabProps {
  userId: number
}

interface TasksTabData {
  tasks: Array<{
    id: number
    title: string
    status: string
    statusName?: string
    projectId: number
    projectKey: string
    projectName: string
    assigneeId?: number | null
    priority?: string
    dueDate?: string | null
    [key: string]: any
  }>
}

export function ProfileTasksTab({ userId }: ProfileTasksTabProps) {
  const { language } = useApp()
  const T = (tr: string, en: string) => language === "tr" ? tr : en
  const [filter, setFilter] = React.useState<TaskFilter>("active")

  const query = useQuery<TasksTabData>({
    queryKey: ["profile-tasks", userId],
    queryFn: () => profileService.getUserTasks(userId),
    enabled: !!userId,
    staleTime: 30_000,
  })

  const allTasks = query.data?.tasks ?? []
  const filtered = React.useMemo(() => allTasks.filter((t) => {
    const isDone = (t.statusName || t.status || "").toLowerCase().includes("done")
      || (t.statusName || t.status || "").toLowerCase().includes("tamamla")
    if (filter === "active") return !isDone
    if (filter === "completed") return isDone
    return true
  }), [allTasks, filter])

  // Group by projectId — preserve insertion order
  const grouped = React.useMemo(() => {
    const map = new Map<number, { projectId: number; projectKey: string; projectName: string; tasks: typeof filtered }>()
    for (const t of filtered) {
      if (!map.has(t.projectId)) {
        map.set(t.projectId, { projectId: t.projectId, projectKey: t.projectKey, projectName: t.projectName, tasks: [] })
      }
      map.get(t.projectId)!.tasks.push(t)
    }
    return Array.from(map.values())
  }, [filtered])

  return (
    <div>
      <div style={{ display: "flex", alignItems: "center", gap: 10, marginBottom: 14 }}>
        <SegmentedControl
          size="sm"
          options={[
            { id: "active", label: T("Aktif", "Active") },
            { id: "completed", label: T("Tamamlanan", "Completed") },
            { id: "all", label: T("Tümü", "All") },
          ]}
          value={filter}
          onChange={(v) => setFilter(v as TaskFilter)}
        />
        <div style={{ flex: 1 }} />
        <span className="mono" style={{ fontSize: 11.5, color: "var(--fg-muted)" }}>
          {filtered.length} {T("görev", "tasks")}
        </span>
      </div>
      <DataState
        loading={query.isLoading}
        error={query.error}
        empty={filtered.length === 0}
        emptyFallback={
          <div style={{ padding: 40, textAlign: "center", color: "var(--fg-subtle)", fontSize: 13 }}>
            {T("Bu filtreyle görev bulunamadı.", "No tasks found with this filter.")}
          </div>
        }
      >
        {grouped.map((group) => (
          <Card key={group.projectId} padding={0} style={{ marginBottom: 12 }}>
            <div
              style={{
                padding: "10px 14px", background: "var(--surface-2)",
                display: "flex", alignItems: "center", gap: 8,
                borderBottom: "1px solid var(--border)",
              }}
            >
              <span style={{ width: 8, height: 8, borderRadius: 3, background: `var(--av-${(group.projectId % 8) + 1})` }} />
              <span className="mono" style={{ fontSize: 10.5, color: "var(--fg-muted)" }}>
                {group.projectKey}
              </span>
              <span style={{ fontSize: 12.5, fontWeight: 600 }}>{group.projectName}</span>
              <Badge size="xs">{group.tasks.length}</Badge>
            </div>
            {group.tasks.map((task) => (
              <TaskRow key={task.id} task={task as any} density="compact" showProject={false} />
            ))}
          </Card>
        ))}
      </DataState>
    </div>
  )
}
```

```tsx
const VALID_TABS: TabId[] = ["tasks", "projects", "activity"]

export default function UserProfilePage() {

  const initialTab: TabId = tabParam && VALID_TABS.includes(tabParam) ? tabParam : "tasks"
  const [activeTab, setActiveTab] = React.useState<TabId>(initialTab)

  // Sync URL on tab change
  const handleTabChange = (id: string) => {
    setActiveTab(id as TabId)
    router.replace(`/users/${userId}?tab=${id}`)
  }

  const { data: summary, isLoading } = useUserSummary(userId)

  if (isLoading) {
    return (
      <div style={{ padding: 40, textAlign: "center", color: "var(--fg-muted)" }}>
        {T("Yükleniyor…", "Loading…")}
      </div>
    )
  }

  if (!summary || !summary.user) {
    return (
      <Card padding={40}>
        <div style={{ textAlign: "center", color: "var(--fg-muted)" }}>
          {T("Kullanıcı bulunamadı.", "User not found.")}
        </div>
      </Card>
    )
  }

  const { user, stats, projects } = summary
  const completionRate = stats.completionRate ?? (
    stats.assignedTasks > 0 ? Math.round((stats.completedTasks / stats.assignedTasks) * 100) : 0
  )

  // Activity tab is mounted in Plan 13-06 — until then we render the Projects stub for the activity slot too,
  // OR if Plan 13-06 has shipped, the import will be augmented to include <ActivityTab userId={userId} variant="full"/>.
  // For Plan 13-05's scope we render a "Yükleniyor…" placeholder inside ActivityTab slot.

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 20, maxWidth: 1100 }}>
      <ProfileHeader user={user} stats={{
        projectsTotal: stats.projectsTotal,
        assignedTasks: stats.assignedTasks,
        completedTasks: stats.completedTasks,
      }} />
      <div className="profile-statcards-grid" style={{ display: "grid", gridTemplateColumns: "repeat(3, 1fr)", gap: 12 }}>
        <StatCard
          label={T("Atanan Görevler", "Assigned Tasks")}
          value={stats.assignedTasks}
          delta={`${stats.assignedTasks} ${T("toplam", "total")}`}
          tone="primary"
          icon={<CheckSquare size={14} />}
        />
        <StatCard
          label={T("Tamamlanan", "Completed")}
          value={stats.completedTasks}
          delta={`%${completionRate} ${T("oran", "rate")}`}
          tone="success"
          icon={<CircleCheck size={14} />}
        />
        <StatCard
          label={T("Projeler", "Projects")}
          value={stats.projectsTotal}
          delta={`${stats.projectsLed} ${T("yönetici", "lead")}`}
          tone="info"
          icon={<Folder size={14} />}
        />
      </div>
      <Tabs
        size="md"
        active={activeTab}
        onChange={handleTabChange}
        tabs={[
          { id: "tasks", label: T("Görevler", "Tasks") },
          { id: "projects", label: T("Projeler", "Projects"), badge: projects?.length ?? 0 },
          { id: "activity", label: T("Aktivite", "Activity") },
        ]}
      />
      {activeTab === "tasks" && <ProfileTasksTab userId={userId} />}
      {activeTab === "projects" && <ProfileProjectsTab userId={userId} />}
      {activeTab === "activity" && (
        // Plan 13-06 will replace this with <ActivityTab userId={userId} variant="full"/>
        <div style={{ padding: 40, textAlign: "center", color: "var(--fg-muted)" }}>
          {T("Yükleniyor…", "Loading…")}
        </div>
      )}
    </div>
  )
}
```
