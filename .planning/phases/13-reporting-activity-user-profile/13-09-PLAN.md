**Step 7 — AvatarDropdown ArrowDown/ArrowUp keyboard navigation:**

```tsx
React.useEffect(() => {
  if (!open) return
  const handler = (e: KeyboardEvent) => {
    if (e.key !== "ArrowDown" && e.key !== "ArrowUp" && e.key !== "Home" && e.key !== "End") return
    if (!ref.current) return
    const items = Array.from(
      ref.current.querySelectorAll<HTMLElement>('[role="menuitem"], [role="menuitemradio"]'),
    )
    if (items.length === 0) return
    const active = document.activeElement as HTMLElement | null
    const idx = active ? items.indexOf(active) : -1
    let nextIdx = idx
    if (e.key === "ArrowDown") nextIdx = idx >= items.length - 1 ? 0 : idx + 1
    else if (e.key === "ArrowUp") nextIdx = idx <= 0 ? items.length - 1 : idx - 1
    else if (e.key === "Home") nextIdx = 0
    else if (e.key === "End") nextIdx = items.length - 1
    items[nextIdx]?.focus()
    e.preventDefault()
  }
  document.addEventListener("keydown", handler)
  return () => document.removeEventListener("keydown", handler)
}, [open])
```
