      try { const r = await fetch("/api/v1/health"); return r.ok } catch { return false }
    }).catch(() => false)

    await expect(timeline.or(empty)).toBeVisible({ timeout: 10_000 })
  })

  test("filter persists across reload (localStorage spms.activity.filter.1)", async ({ page }) => {
    await page.goto("/projects/1?tab=activity")
    // Click Yaşam Döngüsü filter chip
    const chip = page.getByRole("button", { name: /Yaşam Döngüsü|Lifecycle/ }).first()
    if (await chip.isVisible().catch(() => false)) {
      await chip.click()
      const stored = await page.evaluate(() => localStorage.getItem("spms.activity.filter.1"))
      expect(stored).toContain("\"type\":\"lifecycle\"")
      await page.reload()
      const stored2 = await page.evaluate(() => localStorage.getItem("spms.activity.filter.1"))
      expect(stored2).toContain("\"type\":\"lifecycle\"")
    }
  })
})
