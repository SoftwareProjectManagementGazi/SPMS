1.2. Create `Frontend2/lib/activity/group-by-date.ts`:
```ts
import type { ActivityItem } from "@/services/activity-service"
import { formatActivityDate } from "@/lib/activity-date-format"

export interface DateGroup {
  label: string  // "Bugün" / "Dün" / "Bu Hafta" / exact-date
  events: ActivityItem[]
}

export function groupByDate(events: ActivityItem[], language: "tr" | "en"): DateGroup[] {
  // Sort reverse-chronological first
  const sorted = [...events].sort((a, b) => {
    const ta = a.timestamp ? new Date(a.timestamp).getTime() : 0
    const tb = b.timestamp ? new Date(b.timestamp).getTime() : 0
    return tb - ta
  })
  const map = new Map<string, ActivityItem[]>()
  for (const ev of sorted) {
    const label = ev.timestamp ? formatActivityDate(ev.timestamp, language) : (language === "tr" ? "Bilinmeyen" : "Unknown")
    if (!map.has(label)) map.set(label, [])
    map.get(label)!.push(ev)
  }
  // Map preserves insertion order; first inserted = most recent group
  return Array.from(map, ([label, events]) => ({ label, events }))
}
```

```tsx
"use client"
import * as React from "react"
import { SegmentedControl, Avatar } from "@/components/primitives"
import { useApp } from "@/context/app-context"
import type { ActivityItem } from "@/services/activity-service"

export interface ActivityFilterValue {
  type: "all" | "create" | "status" | "assign" | "comment" | "lifecycle"
  userIdFilter: number | null
  showCount: number
}

interface ActivityFilterProps {
  filter: ActivityFilterValue
  setFilter: (next: ActivityFilterValue) => void
  events: ActivityItem[]  // source of unique-actors list
  totalCount: number  // displayed in caption
}

export function ActivityFilter({ filter, setFilter, events, totalCount }: ActivityFilterProps) {
  const { language } = useApp()
  const T = (tr: string, en: string) => language === "tr" ? tr : en

  // Unique actors derived from events list
  const actors = React.useMemo(() => {
    const seen = new Map<number, { id: number; name: string; initials: string }>()
    for (const ev of events) {
      if (ev.user_id == null || ev.user_name == null) continue
      if (!seen.has(ev.user_id)) {
        const initials = ev.user_name.split(" ").filter(Boolean).slice(0, 2).map(n => n[0] || "").join("").toUpperCase()
        seen.set(ev.user_id, { id: ev.user_id, name: ev.user_name, initials })
      }
    }
    return Array.from(seen.values())
  }, [events])

  return (
    <div style={{ display: "flex", alignItems: "center", gap: 10, flexWrap: "wrap" }}>
      <SegmentedControl
        size="sm"
        options={[
          { id: "all", label: T("Tümü", "All") },
          { id: "create", label: T("Oluşturma", "Created") },
          { id: "status", label: T("Durum", "Status") },
          { id: "assign", label: T("Atama", "Assign") },
          { id: "comment", label: T("Yorum", "Comment") },
          { id: "lifecycle", label: T("Yaşam Döngüsü", "Lifecycle") },
        ]}
        value={filter.type}
        onChange={(v) => setFilter({ ...filter, type: v as ActivityFilterValue["type"] })}
      />
      <div style={{ width: 1, height: 20, background: "var(--border)" }} />
      <div style={{ display: "flex", gap: 4 }}>
        {actors.map((a) => (
          <button
            key={a.id}
            type="button"
            onClick={() => setFilter({ ...filter, userIdFilter: filter.userIdFilter === a.id ? null : a.id })}
            aria-label={a.name}
            title={a.name}
            style={{ background: "transparent", border: "none", padding: 0, cursor: "pointer", borderRadius: "50%" }}
          >
            <Avatar user={{ initials: a.initials }} size={24} ring={filter.userIdFilter === a.id} />
          </button>
        ))}
      </div>
      <div style={{ flex: 1 }} />
      <span className="mono" style={{ fontSize: 11, color: "var(--fg-muted)" }}>
        {totalCount} {T("olay", "events")}
      </span>
    </div>
  )
}
```

```tsx
"use client"
import * as React from "react"
import Link from "next/link"
import { useRouter } from "next/navigation"
import { Avatar, Badge } from "@/components/primitives"
import { ArrowRight } from "lucide-react"
import type { ActivityItem } from "@/services/activity-service"
import { mapAuditToSemantic, type SemanticEventType } from "@/lib/audit-event-mapper"
import { eventMeta } from "@/lib/activity/event-meta"
import { formatRelativeTime } from "@/lib/activity-date-format"
import { useApp } from "@/context/app-context"

interface ActivityRowProps {
  event: ActivityItem
  projectId?: number  // when set, taskKey/phase clicks route under that project
}

const COMMENT_PREVIEW_MAX = 160  // D-B6

function statusLabelFromColumnId(value: string | null | undefined, language: "tr" | "en"): string | null {
  // Backend stores column_id; we don't have a column-name lookup at row level.
  // Fall back to the raw value so the timeline renders something rather than empty.
  if (!value) return null
  return value
}

export function ActivityRow({ event, projectId }: ActivityRowProps) {
  const { language } = useApp()
  const router = useRouter()
  const T = (tr: string, en: string) => language === "tr" ? tr : en

  const semantic = mapAuditToSemantic(event)
  if (!semantic) return null

  const meta = eventMeta[semantic]
  const actorInitials = (event.user_name || "")
    .split(" ").filter(Boolean).slice(0, 2).map(n => n[0] || "").join("").toUpperCase()
  const actorAvatar = { initials: actorInitials || "?", avColor: ((event.user_id || 0) % 8) + 1 }
  const firstName = (event.user_name || "").split(" ")[0] || ""

  // task/phase reference label (from entity_label fallback)
  const refLabel = event.entity_label || null

  const taskKey = (event.metadata as any)?.task_key as string | undefined
  const phaseId = (event.metadata as any)?.phase_id as string | undefined
  const taskTitle = (event.metadata as any)?.task_title as string | undefined

  const onRefClick = (e: React.MouseEvent) => {
    e.stopPropagation()
    if (taskKey && projectId != null) router.push(`/projects/${projectId}/tasks/${taskKey}`)
    else if (phaseId && projectId != null) router.push(`/projects/${projectId}?tab=lifecycle&sub=history`)
  }

  return (
    <div
      className="hover-row activity-event-row"
      role="article"
      aria-label={`${event.user_name || ""} ${meta.verb(language)} — ${event.timestamp ? formatRelativeTime(event.timestamp, language) : ""}`}
      style={{ display: "flex", gap: 12, padding: "12px 16px", position: "relative" }}
    >
      {/* Avatar + event badge */}
      <div style={{ position: "relative", zIndex: 1 }}>
        <Avatar
          user={actorAvatar}
          size={28}
          href={event.user_id ? `/users/${event.user_id}` : undefined}
          className="activity-event-avatar"
        />
        <div
          style={{
            position: "absolute", bottom: -2, right: -2, width: 16, height: 16,
            borderRadius: "50%", background: meta.color,
            boxShadow: "0 0 0 2px var(--surface)",
            display: "inline-flex", alignItems: "center", justifyContent: "center",
          }}
          aria-hidden
        >
          <meta.Icon size={9} color="var(--primary-fg)" />
        </div>
      </div>

      {/* Content */}
      <div style={{ flex: 1, minWidth: 0 }}>
        <div style={{ fontSize: 12.5 }}>
          <span style={{ fontWeight: 600 }}>{firstName}</span>
          <span style={{ color: "var(--fg-muted)" }}> {meta.verb(language)} </span>
          {refLabel && (
            <span
              style={{ color: "var(--primary)", cursor: "pointer", fontWeight: 500 }}
              onClick={onRefClick}
            >
              {refLabel}
            </span>
          )}
        </div>
        {taskTitle && (
          <div style={{ fontSize: 12, color: "var(--fg-muted)", marginTop: 2 }}>
            {taskTitle}
          </div>
        )}
        {/* Status pair (task_status_changed) */}
        {semantic === "task_status_changed" && (
          <div style={{ display: "flex", gap: 6, alignItems: "center", marginTop: 6 }}>
            {event.old_value && <Badge size="xs" tone="neutral">{statusLabelFromColumnId(event.old_value, language)}</Badge>}
            <ArrowRight size={11} style={{ color: "var(--fg-muted)" }} />
            {event.new_value && <Badge size="xs" tone="info">{statusLabelFromColumnId(event.new_value, language)}</Badge>}
          </div>
        )}
        {/* Assign target row */}
        {semantic === "task_assigned" && event.new_value && (
          <div style={{ display: "flex", gap: 6, alignItems: "center", marginTop: 6, fontSize: 12 }}>
            <ArrowRight size={11} style={{ color: "var(--fg-muted)" }} />
            <Avatar user={{ initials: "?" }} size={18} />
            <span>{event.new_value}</span>
          </div>
        )}
        {/* Comment block (D-B6 — 160 chars + ellipsis, tinted block) */}
        {semantic === "comment_created" && event.new_value && (
          <div
            style={{

            })()}

          {event.timestamp ? formatRelativeTime(event.timestamp, language) : ""}

    </div>
  )
```

```tsx
"use client"
import * as React from "react"
import { Card, DataState, Button } from "@/components/primitives"
import { useProjectActivity } from "@/hooks/use-project-activity"
import { useUserActivity } from "@/hooks/use-user-activity"
import { useLocalStoragePref } from "@/hooks/use-local-storage-pref"
import { mapAuditToSemantic, semanticToFilterChip } from "@/lib/audit-event-mapper"
import { groupByDate } from "@/lib/activity/group-by-date"
import { useApp } from "@/context/app-context"
import { ActivityRow } from "./activity-row"
import { ActivityFilter, type ActivityFilterValue } from "./activity-filter"
import { ActivityEmpty } from "./activity-empty"
import { ActivityTimelineSkeleton } from "./activity-skeleton"

type ActivityTabProps =
  | { projectId: number; userId?: never; variant?: "full" | "compact" }
  | { projectId?: never; userId: number; variant?: "full" | "compact" }

const DEFAULT_FILTER: ActivityFilterValue = { type: "all", userIdFilter: null, showCount: 30 }

export function ActivityTab(props: ActivityTabProps) {
  const variant = props.variant ?? "full"
  const { language } = useApp()
  const T = (tr: string, en: string) => language === "tr" ? tr : en

  // localStorage key per project / per user (D-B7)
  const storageKey = props.projectId
    ? `activity.filter.${props.projectId}`
    : `activity.filter.user.${props.userId}`
  const [filter, setFilter] = useLocalStoragePref<ActivityFilterValue>(storageKey, DEFAULT_FILTER)

  // Build the network filter (only `type[]` and limit go to the server; client-side filters apply locally)
  const serverFilter = React.useMemo(() => ({ limit: 200, offset: 0 }), [])

  // Discriminated routing — call BOTH hooks but enable only one (D-B4)
  const projectQuery = useProjectActivity(props.projectId, serverFilter)
  const userQuery = useUserActivity(props.userId, serverFilter)
  const query = props.projectId ? projectQuery : userQuery

  const allEvents = query.data?.items ?? []

  // Apply client-side filtering: type chip + actor filter
  const filtered = React.useMemo(() => {
    return allEvents.filter((ev) => {
      const semantic = mapAuditToSemantic(ev)
      if (!semantic) return false
      const chip = semanticToFilterChip(semantic)
      if (filter.type !== "all" && chip !== filter.type) return false
      if (filter.userIdFilter != null && ev.user_id !== filter.userIdFilter) return false
      return true
    })
  }, [allEvents, filter.type, filter.userIdFilter])

  const visible = filtered.slice(0, filter.showCount)
  const grouped = React.useMemo(() => groupByDate(visible, language), [visible, language])

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 14 }}>
      <ActivityFilter filter={filter} setFilter={setFilter} events={allEvents} totalCount={filtered.length} />
      <Card padding={0}>
        <div className="activity-timeline" style={{ position: "relative", paddingLeft: 32 }}>
          <div
            className="activity-timeline-line"
            style={{ position: "absolute", left: 19, top: 20, bottom: 20, width: 2, background: "var(--border)" }}
            aria-hidden
          />
          <DataState
            loading={query.isLoading}
            loadingFallback={<ActivityTimelineSkeleton />}
            error={query.error}
            empty={visible.length === 0}
            emptyFallback={<ActivityEmpty filtered={filter.type !== "all" || filter.userIdFilter != null} />}
          >
            {grouped.map((group) => (
              <div key={group.label}>
                <div style={{
                  padding: "14px 16px 8px", fontSize: 11, fontWeight: 600,
                  color: "var(--fg-subtle)", textTransform: "uppercase", letterSpacing: 0.5,
                }}>
                  {group.label}
                </div>
                {group.events.map((ev) => (
                  <ActivityRow key={ev.id} event={ev} projectId={props.projectId} />
                ))}
              </div>
            ))}
            {filtered.length > filter.showCount && (
              <div style={{ padding: 16, textAlign: "center" }}>
                <Button
                  variant="ghost"
                  size="sm"
                  onClick={() => setFilter({ ...filter, showCount: filter.showCount + 30 })}
                >
                  {T("Daha fazla yükle", "Load more")}
                </Button>
              </div>
            )}
          </DataState>
        </div>
      </Card>
    </div>
  )
}
```
