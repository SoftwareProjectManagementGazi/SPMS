```typescript
"use client"

import { PermissionScopeBadge } from "./permission-scope-badge"
import { useUpdatePermissionCell } from "@/hooks/use-update-permission-cell"
import type { Permission, Role, MatrixCell } from "@/services/admin-rbac-service"

interface PermissionRowProps {
  permission: Permission           // includes scope field per Plan 15-09 service
  roles: Role[]
  cells: MatrixCell[]              // global matrix cells; component filters per (role.id, permission.id)
}

export function PermissionRow({ permission, roles, cells }: PermissionRowProps) {
  const updateCell = useUpdatePermissionCell()

  const isGranted = (roleId: number) =>
    cells.some((c) => c.role_id === roleId && c.permission_id === permission.id && c.granted)

  return (
    <tr>
      <td>
        <span style={{ display: "inline-flex", gap: 8, alignItems: "center" }}>
          {permission.label_tr || permission.key}
          <PermissionScopeBadge scope={permission.scope} />
        </span>
      </td>
      {roles.map((role) => {
        const isAdminColumn = role.name.toLowerCase() === "admin"
        const granted = isGranted(role.id)
        return (
          <td key={role.id}>
            <input
              type="checkbox"
              role="switch"
              checked={granted || isAdminColumn /* D-1.5 super-role visual indicator */}
              disabled={isAdminColumn || role.name.toLowerCase() === "guest" /* D-2.4 read-only */ || updateCell.isPending}
              aria-disabled={isAdminColumn || role.name.toLowerCase() === "guest"}
              onChange={(e) => {
                if (isAdminColumn || role.name.toLowerCase() === "guest") return
                updateCell.mutate({
                  roleId: role.id,
                  permKey: permission.key,
                  granted: e.target.checked,
                })
              }}
              aria-label={`${permission.key} for ${role.name}`}
            />
          </td>
        )
      })}
    </tr>
  )
}
```

KEY REQUIREMENT: NO `v3.0` text anywhere. NO blanket disabled attr (only Admin + Guest columns retain disabled — Admin is super-role visual; Guest is read-only system role per D-2.4).
