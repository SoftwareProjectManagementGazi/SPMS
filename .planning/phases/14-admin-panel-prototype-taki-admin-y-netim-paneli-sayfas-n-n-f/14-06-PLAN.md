    ```tsx
    "use client"
    import { Card, Badge } from "@/components/primitives"
    import { TemplateRowActions } from "./template-row-actions"
    import { useApp } from "@/context/app-context"
    import { adminT } from "@/lib/i18n/admin-keys"

    export function AdminTemplateCard({ template }: { template: ProcessTemplate }) {
      const { language } = useApp()
      const modeTone = template.mode === "sequential-locked" ? "warning" : template.mode === "continuous" ? "success" : "neutral"
      return (
        <Card padding={16}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: 8 }}>
            <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
              <h3 style={{ fontSize: 14, fontWeight: 600, margin: 0 }}>{template.name}</h3>
              {template.is_custom && <Badge tone="primary" size="xs">{adminT("admin.workflows.custom_badge", language)}</Badge>}
            </div>
            <TemplateRowActions template={template} />
          </div>
          {template.description && (
            <p style={{ fontSize: 12, lineHeight: 1.5, color: "var(--fg-muted)", margin: "0 0 12px 0" }}>
              {template.description}
            </p>
          )}
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
            <Badge tone={modeTone} size="sm">{template.mode}</Badge>
            <span style={{ fontSize: 11.5, color: "var(--fg-muted)" }}>
              {template.active_project_count ?? 0} {adminT("admin.workflows.proje_suffix", language)}
            </span>
          </div>
        </Card>
      )
    }
    ```

    Step 4 — Write template-row-actions.tsx with 3-item MoreMenu. For Sil:
    ```tsx
    const [confirmOpen, setConfirmOpen] = useState(false)
    const [yineDe, setYineDe] = useState(false)
    const inUse = (template.active_project_count ?? 0) > 0
    const canDelete = !inUse || yineDe

    return (
      <>
        <MoreMenu items={[
          { id: "edit", label: adminT("admin.workflows.edit", language), onClick: () => router.push(`/workflow-editor?templateId=${template.id}`) },
          { id: "clone", label: adminT("admin.workflows.clone", language), onClick: () => cloneTemplate.mutate(template.id) },
          { id: "delete", label: adminT("admin.workflows.delete", language), onClick: () => setConfirmOpen(true), destructive: true },
        ]}/>
        {confirmOpen && (
          <ConfirmDialog
            open={confirmOpen}
            tone="danger"
            title={adminT("admin.workflows.delete_modal_title", language).replace("{name}", template.name)}
            body={
              inUse
                ? adminT("admin.workflows.delete_modal_body_in_use", language)
                    .replace("{name}", template.name).replace("{count}", String(template.active_project_count))
                : adminT("admin.workflows.delete_modal_body", language).replace("{name}", template.name)
            }
            confirmLabel={adminT("admin.workflows.delete", language)}
            onConfirm={canDelete ? () => { deleteTemplate.mutate(template.id); setConfirmOpen(false); setYineDe(false) } : () => {}}
            onCancel={() => { setConfirmOpen(false); setYineDe(false) }}
          />
        )}
        {/* For inUse case: render an additional Checkbox below ConfirmDialog body via a custom Modal wrapper */}
      </>
    )
    ```
