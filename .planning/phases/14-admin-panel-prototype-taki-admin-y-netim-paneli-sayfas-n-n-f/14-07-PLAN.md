    "use client"
    import { useSearchParams, useRouter, usePathname } from "next/navigation"
    import dynamic from "next/dynamic"
    import { AdminAuditToolbar } from "@/components/admin/audit/admin-audit-toolbar"
    import { AuditFilterChips } from "@/components/admin/audit/audit-filter-chips"
    import { DataState } from "@/components/primitives"

    const AdminAuditTable = dynamic(() => import("@/components/admin/audit/admin-audit-table").then(m => m.AdminAuditTable), {
      ssr: false,
      loading: () => <DataState loading />,
    })

    export default function AdminAuditPage() {
      const searchParams = useSearchParams()
      const router = useRouter()
      const pathname = usePathname()
      const filter = parseFilterFromParams(searchParams)

      const updateFilter = (next: Partial<typeof filter>) => {
        const merged = { ...filter, ...next, offset: next.offset ?? 0 }  // D-Z2 reset to 0 on filter change unless offset explicitly set
        router.replace(`${pathname}?${filterToParams(merged).toString()}`)
      }

      return (
        <div style={{ display: "flex", flexDirection: "column", gap: 16 }}>
          <AdminAuditToolbar filter={filter} onUpdate={updateFilter} />
          <AuditFilterChips filter={filter} onUpdate={updateFilter} />
          <AdminAuditTable filter={filter} />
          {/* Pagination is rendered inside AdminAuditTable to access response.total */}
        </div>
      )
    }

    function parseFilterFromParams(p: URLSearchParams): AdminAuditFilter {
      return {
        date_from: p.get("from") ?? undefined,
        date_to: p.get("to") ?? undefined,
        actor_id: p.get("actor_id") ? parseInt(p.get("actor_id")!) : undefined,
        action_prefix: p.get("action_prefix") ?? undefined,
        limit: p.get("size") ? parseInt(p.get("size")!) : 50,
        offset: p.get("offset") ? parseInt(p.get("offset")!) : 0,
      }
    }
    function filterToParams(f: AdminAuditFilter): URLSearchParams {
      const p = new URLSearchParams()
      if (f.date_from) p.set("from", f.date_from)
      if (f.date_to) p.set("to", f.date_to)
      if (f.actor_id) p.set("actor_id", String(f.actor_id))
      if (f.action_prefix) p.set("action_prefix", f.action_prefix)
      if (f.limit && f.limit !== 50) p.set("size", String(f.limit))
      if (f.offset && f.offset !== 0) p.set("offset", String(f.offset))
      return p
    }

    "use client"
    import { useState } from "react"
    import { Modal, ModalHeader, ModalBody, ModalFooter, Input, Button } from "@/components/primitives"
    import { useApp } from "@/context/app-context"
    import { useAdminUsers } from "@/hooks/use-admin-users"
    import { adminT } from "@/lib/i18n/admin-keys"
    import type { AdminAuditFilter } from "@/services/admin-audit-service"

    interface Props {
      open: boolean
      filter: AdminAuditFilter
      onApply: (next: AdminAuditFilter) => void
      onClose: () => void
    }

    export function AuditFilterModal({ open, filter, onApply, onClose }: Props) {
      const { language } = useApp()
      const { data: users } = useAdminUsers({})
      const [draft, setDraft] = useState<AdminAuditFilter>(filter)

      // Reset draft when modal opens
      React.useEffect(() => { if (open) setDraft(filter) }, [open, filter])

      return (
        <Modal open={open} onClose={onClose} width={520}>
          <ModalHeader>{adminT("admin.audit.filter_modal_title", language)}</ModalHeader>
          <ModalBody>
            <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
              <label style={{ fontSize: 12, color: "var(--fg-muted)" }}>
                {adminT("admin.audit.filter_date_from", language)}
                <Input type="date" value={draft.date_from?.slice(0, 10) ?? ""}
                  onChange={e => setDraft({ ...draft, date_from: e.target.value || undefined })} />
              </label>
              <label style={{ fontSize: 12, color: "var(--fg-muted)" }}>
                {adminT("admin.audit.filter_date_to", language)}
                <Input type="date" value={draft.date_to?.slice(0, 10) ?? ""}
                  onChange={e => setDraft({ ...draft, date_to: e.target.value || undefined })} />
              </label>
              <label style={{ fontSize: 12, color: "var(--fg-muted)" }}>
                {adminT("admin.audit.filter_actor", language)}
                <select value={draft.actor_id ?? ""}
                  onChange={e => setDraft({ ...draft, actor_id: e.target.value ? parseInt(e.target.value) : undefined })}>
                  <option value="">{adminT("admin.users.filter_all", language)}</option>
                  {users?.items.map(u => <option key={u.id} value={u.id}>{u.email}</option>)}
                </select>
              </label>
              <label style={{ fontSize: 12, color: "var(--fg-muted)" }}>
                {adminT("admin.audit.filter_action_prefix", language)}
                <Input type="text" value={draft.action_prefix ?? ""}
                  placeholder="task. veya project.archive"
                  onChange={e => setDraft({ ...draft, action_prefix: e.target.value || undefined })} />
              </label>
            </div>
          </ModalBody>
          <ModalFooter>
            <Button variant="ghost" onClick={onClose}>{adminT("admin.audit.filter_cancel", language)}</Button>
            <Button variant="ghost" onClick={() => { onApply({}); onClose() }}>{adminT("admin.audit.filter_clear", language)}</Button>
            <Button variant="primary" onClick={() => { onApply(draft); onClose() }}>{adminT("admin.audit.filter_apply", language)}</Button>
          </ModalFooter>
        </Modal>
      )
    }
