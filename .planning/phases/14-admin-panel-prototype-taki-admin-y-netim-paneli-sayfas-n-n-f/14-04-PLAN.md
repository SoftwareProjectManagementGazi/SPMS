    ```tsx
    "use client"
    import { Toggle } from "@/components/primitives"
    import { useApp } from "@/context/app-context"
    import { adminT } from "@/lib/i18n/admin-keys"
    import { getPermission, type PermissionRow as PermRow, type PermissionRole } from "@/lib/admin/permissions-static"

    const ROLES: PermissionRole[] = ["Admin", "Project Manager", "Member", "Guest"]

    export function PermissionRow({ perm }: { perm: PermRow }) {
      const { language } = useApp()
      const tooltip = adminT("admin.permissions.toggle_tooltip", language)
      return (
        <div style={{
          display: "grid",
          gridTemplateColumns: "2fr repeat(4, 100px)",
          padding: "8px 16px", alignItems: "center",
          borderBottom: "1px solid var(--border)",
        }}>
          <div style={{ fontSize: 12.5, color: "var(--fg)" }}>{perm.label[language]}</div>
          {ROLES.map(role => {
            const state = getPermission(perm.id, role)
            return (
              <div key={role} style={{ display: "flex", justifyContent: "center" }}>
                <Toggle
                  size="sm"
                  checked={state === "granted"}
                  disabled
                  ariaLabel={`${perm.label[language]} — ${role} — disabled`}
                />
              </div>
            )
          })}
        </div>
      )
    }
    ```
