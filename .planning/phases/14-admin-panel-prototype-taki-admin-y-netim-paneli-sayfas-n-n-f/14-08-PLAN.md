    ```tsx
    import dynamic from "next/dynamic"
    import { useAdminStats } from "@/hooks/use-admin-stats"

    const ActiveUsersTrendChart = dynamic(() => import("@/components/admin/stats/active-users-trend-chart").then(m => m.ActiveUsersTrendChart), { ssr: false, loading: () => <DataState loading /> })
    const MethodologyBars = dynamic(() => import("@/components/admin/stats/methodology-bars").then(m => m.MethodologyBars), { ssr: false, loading: () => <DataState loading /> })
    const VelocityCardsGrid = dynamic(() => import("@/components/admin/stats/velocity-cards-grid").then(m => m.VelocityCardsGrid), { ssr: false, loading: () => <DataState loading /> })

    export default function AdminStatsPage() {
      const { data, isLoading, error } = useAdminStats()
      if (isLoading) return <DataState loading />
      if (error) return <DataState error={error.message} />
      if (!data) return <DataState empty="No stats available" />

      return (
        <div style={{ display: "grid", gap: 20 }}>
          <div style={{ display: "grid", gridTemplateColumns: "2fr 1fr", gap: 20 }}>
            <ActiveUsersTrendChart trend={data.activeUsersTrend} />
            <MethodologyBars distribution={data.methodologyDistribution} />
          </div>
          <VelocityCardsGrid velocities={data.projectVelocities} />
        </div>
      )
    }
    ```

    Step 2 — Write active-users-trend-chart.tsx. Use recharts:
    ```tsx
    "use client"
    import { ResponsiveContainer, AreaChart, Area, XAxis, YAxis, Tooltip } from "recharts"
    import { Card, Badge } from "@/components/primitives"
    import { useApp } from "@/context/app-context"
    import { adminT } from "@/lib/i18n/admin-keys"

    export function ActiveUsersTrendChart({ trend }: { trend: { date: string; count: number }[] }) {
      const { language } = useApp()
      const first = trend[0]?.count ?? 0
      const last = trend[trend.length - 1]?.count ?? 0
      const deltaPct = first > 0 ? Math.round(((last - first) / first) * 100) : 0

      return (
        <Card padding={16}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: 12 }}>
            <div>
              <h3 style={{ fontSize: 13, fontWeight: 600, margin: 0 }}>{adminT("admin.stats.active_users_title", language)}</h3>
              <div style={{ fontSize: 11.5, color: "var(--fg-muted)", marginTop: 4 }}>{adminT("admin.stats.active_users_subtitle", language)}</div>
            </div>
            <Badge tone={deltaPct >= 0 ? "success" : "danger"} size="sm">{deltaPct >= 0 ? "+" : ""}{deltaPct}%</Badge>
          </div>
          {trend.length === 0 ? (
            <div style={{ fontSize: 12, color: "var(--fg-muted)", textAlign: "center", padding: 40 }}>
              {adminT("admin.stats.empty_trend", language)}
            </div>
          ) : (
            <ResponsiveContainer width="100%" height={200}>
              <AreaChart data={trend} margin={{ top: 0, right: 0, left: 0, bottom: 0 }}>
                <defs>
                  <linearGradient id="trendGradient" x1="0" y1="0" x2="0" y2="1">
                    <stop offset="0%" stopColor="var(--primary)" stopOpacity={0.12} />
                    <stop offset="100%" stopColor="var(--primary)" stopOpacity={0} />
                  </linearGradient>
                </defs>
                <XAxis dataKey="date" hide />
                <YAxis hide />
                <Tooltip
                  formatter={(value: any) => [`${value} ${adminT("admin.stats.tooltip_active_users_suffix", language)}`, ""]}
                  labelFormatter={(label: any) => label}
                  contentStyle={{ background: "var(--surface)", border: "1px solid var(--border)", borderRadius: 4, fontSize: 12 }}
                />
                <Area type="monotone" dataKey="count" stroke="var(--primary)" strokeWidth={2} fill="url(#trendGradient)" />
              </AreaChart>
            </ResponsiveContainer>
          )}
        </Card>
      )
    }
    ```

    Step 3 — Write methodology-bars.tsx. Pure CSS bars NO recharts:
    ```tsx
    "use client"
    import { Card } from "@/components/primitives"
    import { useApp } from "@/context/app-context"
    import { adminT } from "@/lib/i18n/admin-keys"

    const COLORS: Record<string, string> = {
      scrum: "var(--status-progress)",
      kanban: "var(--fg-muted)",
      waterfall: "var(--status-review)",
      iterative: "var(--status-done)",
    }

    export function MethodologyBars({ distribution }: { distribution: Record<string, number> }) {
      const { language } = useApp()
      const total = Object.values(distribution).reduce((a, b) => a + b, 0) || 1
      const entries = Object.entries(distribution)

      return (
        <Card padding={16}>
          <h3 style={{ fontSize: 13, fontWeight: 600, marginBottom: 12 }}>{adminT("admin.stats.methodology_title", language)}</h3>
          <div style={{ display: "flex", flexDirection: "column", gap: 10 }}>
            {entries.map(([method, count]) => {
              const pct = (count / total) * 100
              return (
                <div key={method} style={{ display: "grid", gridTemplateColumns: "100px 1fr 36px", alignItems: "center", gap: 10, fontSize: 11.5 }}>
                  <span style={{ textTransform: "capitalize" }}>{method}</span>
                  <div style={{ height: 8, background: "var(--surface-2)", borderRadius: 4, overflow: "hidden" }}>
                    <div style={{ width: `${pct}%`, height: "100%", background: COLORS[method] ?? "var(--fg-muted)" }} />
                  </div>
                  <span style={{ fontFamily: "var(--font-mono)", fontVariantNumeric: "tabular-nums", textAlign: "right" }}>{count}</span>
                </div>
              )
            })}
          </div>
        </Card>
      )
    }
    ```

    Step 1 — Write velocity-mini-bar.tsx:
    ```tsx
    "use client"
    import { Card } from "@/components/primitives"

    interface Velocity {
      projectId: number
      key: string
      name: string
      progress: number  // 0..1
      velocityHistory: number[]  // up to 7 bars
    }

    export function VelocityMiniBar({ v }: { v: Velocity }) {
      const max = Math.max(...v.velocityHistory, 1)
      return (
        <Card padding={12} style={{ display: "flex", flexDirection: "column", gap: 8 }}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "baseline" }}>
            <span style={{ fontFamily: "var(--font-mono)", fontSize: 10.5, fontWeight: 600 }}>{v.key}</span>
            <span style={{ fontSize: 11.5, color: "var(--fg-muted)", overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap", maxWidth: 100 }}>{v.name}</span>
          </div>
          <div style={{ display: "flex", alignItems: "flex-end", gap: 2, height: 32 }}>
            {v.velocityHistory.slice(-7).map((value, i, arr) => {
              const isLast = i === arr.length - 1
              const pct = (value / max) * 100
              return (
                <div key={i} style={{ flex: 1, height: `${pct}%`, background: isLast ? "var(--primary)" : "var(--border-strong)", borderRadius: 2 }} />
              )
            })}
          </div>
          <div style={{ fontSize: 20, fontWeight: 600, letterSpacing: -0.5, fontVariantNumeric: "tabular-nums" }}>
            {Math.round(v.progress * 100)}%
          </div>
        </Card>
      )
    }
    ```
