    ```python
    # Backend/app/infrastructure/database/repositories/audit_repo.py
    # Plan 14-16 (Cluster D, Path B) — entity_label resolver.
    # Priority order:
    #   1. extra_metadata.task_title       (task entity, Plan 14-09 enriched)
    #   2. extra_metadata.project_name     (project entity)
    #   3. extra_metadata.milestone_title  (milestone entity)
    #   4. extra_metadata.artifact_name    (artifact entity)
    #   5. extra_metadata.comment_excerpt  (comment entity — first 60 chars)
    #   6. f"{entity_type.upper()}-{entity_id}"  (legacy fallback for pre-Plan-14-09 rows)
    def _resolve_entity_label(row) -> Optional[str]:
        md = row.get("extra_metadata") or {}
        if not isinstance(md, dict):
            return None
        for key in ("task_title", "project_name", "milestone_title", "artifact_name"):
            if md.get(key):
                return str(md[key])
        if md.get("comment_excerpt"):
            return f"yorum: {str(md['comment_excerpt'])[:60]}"
        if row.get("entity_type") and row.get("entity_id"):
            return f"{row['entity_type'].upper()}-{row['entity_id']}"
        return None
    ```
